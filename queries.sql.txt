SELECT * FROM Students;

SELECT * FROM Courses;

SELECT * FROM Enrollments;

SELECT Students.name, Courses.course_name
FROM Students
JOIN Enrollments
ON Students.student_id = Enrollments.student_id
JOIN Courses
ON Enrollments.course_id = Courses.course_id;

SELECT Students.name, marks
FROM Students
JOIN Enrollments
ON Students.student_id = Enrollments.student_id
WHERE marks > 80;

SELECT AVG(marks)
FROM Enrollments;
